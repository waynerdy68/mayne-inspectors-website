            <button id="close-ai-modal-button" class="absolute top-4 right-4 text-slate-500 hover:text-slate-700 dark:text-slate-400 dark:hover:text-slate-200 transition-colors">
                <i data-lucide="x" class="h-6 w-6"></i>
            </button>
            <h3 class="text-2xl font-semibold text-light-text dark:text-dark-text mb-1 flex items-center">
                <i data-lucide="sparkles" class="h-6 w-6 mr-2 text-primary animate-pulse-fast"></i>
                Ask Our AI Inspector
            </h3>
            <p class="text-sm text-slate-500 dark:text-slate-400 mb-6">Get quick answers to common home inspection questions.</p>
            
            <div class="space-y-4">
                <div>
                    <label for="ai-question-input" class="block text-sm font-medium text-light-text dark:text-dark-text mb-1">Your Question:</label>
                    <textarea id="ai-question-input" rows="3" class="w-full px-4 py-2.5 rounded-md bg-light-bg dark:bg-dark-bg border border-slate-300 dark:border-slate-600 focus:ring-2 focus:ring-primary focus:border-primary outline-none transition-all" placeholder="e.g., What's included in a 4-point inspection?"></textarea>
                </div>
                <button id="submit-ai-question-button" class="w-full bg-primary hover:bg-primary-dark text-white font-semibold py-2.5 px-6 rounded-lg transition-colors duration-300 shadow-md hover:shadow-lg flex items-center justify-center">
                    <span id="ai-button-text">Ask Question</span>
                    <div id="ai-loading-spinner" class="spinner ml-2 hidden"></div>
                </button>
            </div>

            <div id="ai-response-area" class="mt-6">
                <!-- AI response will be injected here -->
                <p class="text-sm text-slate-500 dark:text-slate-400 italic">The AI's answer will appear here. Please be aware that this is an AI assistant and its advice should not be considered a substitute for a professional inspection or consultation.</p>
            </div>
        </div>
    </div>


    <script>
        // Initialize Lucide Icons
        lucide.createIcons();

        // --- Theme Toggler ---
        const themeToggle = document.getElementById('theme-toggle');
        const htmlElement = document.documentElement;
        const sunIcon = themeToggle.querySelector('.theme-icon-sun');
        const moonIcon = themeToggle.querySelector('.theme-icon-moon');

        // Apply saved theme on load
        if (localStorage.getItem('theme') === 'light' || (!('theme' in localStorage) && window.matchMedia('(prefers-color-scheme: light)').matches)) {
            htmlElement.classList.remove('dark');
            sunIcon.classList.add('hidden');
            moonIcon.classList.remove('hidden');
        } else {
            htmlElement.classList.add('dark');
            sunIcon.classList.remove('hidden');
            moonIcon.classList.add('hidden');
        }

        themeToggle.addEventListener('click', () => {
            htmlElement.classList.toggle('dark');
            if (htmlElement.classList.contains('dark')) {
                localStorage.setItem('theme', 'dark');
                sunIcon.classList.remove('hidden');
                moonIcon.classList.add('hidden');
            } else {
                localStorage.setItem('theme', 'light');
                sunIcon.classList.add('hidden');
                moonIcon.classList.remove('hidden');
            }
            lucide.createIcons(); // Re-render icons if theme changes how they might appear
        });

        // --- Mobile Menu ---
        const mobileMenuButton = document.getElementById('mobile-menu-button');
        const mobileMenu = document.getElementById('mobile-menu');
        const mobileNavLinks = mobileMenu.querySelectorAll('.mobile-nav-link');

        mobileMenuButton.addEventListener('click', () => {
            mobileMenu.classList.toggle('hidden');
            const menuIcon = mobileMenuButton.querySelector('i');
            if (mobileMenu.classList.contains('hidden')) {
                menuIcon.setAttribute('data-lucide', 'menu');
            } else {
                menuIcon.setAttribute('data-lucide', 'x');
            }
            lucide.createIcons(); 
        });
        
        mobileNavLinks.forEach(link => {
            link.addEventListener('click', () => {
                mobileMenu.classList.add('hidden');
                mobileMenuButton.querySelector('i').setAttribute('data-lucide', 'menu');
                lucide.createIcons();
            });
        });


        // --- Back to Top Button ---
        const backToTopButton = document.getElementById('back-to-top');
        window.addEventListener('scroll', () => {
            if (window.pageYOffset > 300) {
                backToTopButton.classList.remove('opacity-0', 'translate-y-4');
                backToTopButton.classList.add('opacity-100', 'translate-y-0');
            } else {
                backToTopButton.classList.remove('opacity-100', 'translate-y-0');
                backToTopButton.classList.add('opacity-0', 'translate-y-4');
            }
        });
        backToTopButton.addEventListener('click', () => {
            window.scrollTo({ top: 0, behavior: 'smooth' });
        });

        // --- Scroll-triggered Animations ---
        const animatedElements = document.querySelectorAll('.animate-on-scroll');
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('is-visible');
                }
            });
        }, { threshold: 0.1 });
        animatedElements.forEach(el => { observer.observe(el); });
        

        // --- Form Submission Handling (Placeholder) ---
        function handleFormSubmit(formId, messageDivId, successMessage) {
            const form = document.getElementById(formId);
            const messageDiv = document.getElementById(messageDivId);
            if (form && messageDiv) {
                form.addEventListener('submit', function(event) {
                    event.preventDefault(); 
                    let isValid = true;
                    form.querySelectorAll('[required]').forEach(input => {
                        if (!input.value.trim()) {
                            isValid = false;
                            input.classList.add('border-red-500');
                            input.classList.remove('border-slate-300', 'dark:border-slate-600');
                        } else {
                            input.classList.remove('border-red-500');
                            input.classList.add('border-slate-300', 'dark:border-slate-600');
                        }
                    });

                    if (!isValid) {
                        messageDiv.innerHTML = `<p class="text-red-500 dark:text-red-400">Please fill out all required fields.</p>`;
                        setTimeout(() => { messageDiv.innerHTML = ''; }, 5000);
                        return;
                    }
                    messageDiv.innerHTML = `<p class="text-green-500 dark:text-green-400">${successMessage}</p>`;
                    form.reset(); 
                    setTimeout(() => { messageDiv.innerHTML = ''; }, 5000);
                });
            }
        }
        handleFormSubmit('schedule-form', 'schedule-form-message', 'Thank you! Your inspection request has been submitted. We will contact you shortly.');
        handleFormSubmit('contact-form', 'contact-form-message', 'Thank you for your message! We will get back to you soon.');

        // --- Update current year in footer ---
        document.getElementById('current-year').textContent = new Date().getFullYear();

        // --- AI Assistant Modal Logic ---
        const openAiButton = document.getElementById('open-ai-assistant-button');
        const aiModal = document.getElementById('ai-assistant-modal');
        const closeAiModalButton = document.getElementById('close-ai-modal-button');
        const submitAiQuestionButton = document.getElementById('submit-ai-question-button');
        const aiQuestionInput = document.getElementById('ai-question-input');
        const aiResponseArea = document.getElementById('ai-response-area');
        const aiButtonText = document.getElementById('ai-button-text');
        const aiLoadingSpinner = document.getElementById('ai-loading-spinner');

        if (openAiButton && aiModal && closeAiModalButton && submitAiQuestionButton && aiQuestionInput && aiResponseArea) {
            openAiButton.addEventListener('click', () => {
                aiModal.classList.add('active');
                lucide.createIcons(); // Ensure icons in modal are rendered
            });

            closeAiModalButton.addEventListener('click', () => {
                aiModal.classList.remove('active');
            });

            // Close modal if clicking outside the content
            aiModal.addEventListener('click', (event) => {
                if (event.target === aiModal) {
                    aiModal.classList.remove('active');
                }
            });

            submitAiQuestionButton.addEventListener('click', async () => {
                const userQuestion = aiQuestionInput.value.trim();
                if (!userQuestion) {
                    aiResponseArea.innerHTML = `<p class="text-red-500 dark:text-red-400">Please enter a question.</p>`;
                    return;
                }

                // Show loading state
                aiButtonText.textContent = 'Thinking...';
                aiLoadingSpinner.classList.remove('hidden');
                submitAiQuestionButton.disabled = true;
                aiResponseArea.innerHTML = `<p class="text-slate-500 dark:text-slate-400 italic">Our AI Inspector is generating an answer...</p>`;

                const currentDate = new Date().toLocaleDateString('en-US', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });
                const prompt = `You are a friendly and helpful AI assistant for Mayne Home Inspectors, a home inspection company based in LaBelle, Florida, and serving Southwest Florida (including Lehigh Acres, Clewiston, Moore Haven, Immokalee, Fort Myers, Cape Coral, Punta Gorda, Port Charlotte). Your goal is to answer common questions about home inspections, our services, and general home maintenance tips relevant to Florida.

                Keep your answers concise, informative, and easy to understand.
                Do NOT provide specific pricing, schedule inspections, or give legal/financial advice. For pricing or scheduling, politely direct users to the contact form, scheduling section on the website, or to call Mayne Home Inspectors directly.
                If a question is outside your scope (e.g., asking for medical advice, unrelated topics), politely state that you can only assist with home inspection related queries for Mayne Home Inspectors.
                Today's date is ${currentDate}. The current location is LaBelle, Florida.

                User's question: "${userQuestion}"

                Provide your answer:`;

                try {
                    let chatHistory = [{ role: "user", parts: [{ text: prompt }] }];
                    const payload = { contents: chatHistory };
                    const apiKey = ""; // Canvas will provide this
                    const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;
                    
                    const response = await fetch(apiUrl, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });

                    if (!response.ok) {
                        const errorData = await response.json();
                        console.error('Gemini API Error:', errorData);
                        throw new Error(`API request failed with status ${response.status}. Check console for details.`);
                    }

                    const result = await response.json();

                    if (result.candidates && result.candidates.length > 0 &&
                        result.candidates[0].content && result.candidates[0].content.parts &&
                        result.candidates[0].content.parts.length > 0) {
                        const rawText = result.candidates[0].content.parts[0].text;
                        // Basic Markdown-like formatting for paragraphs (replace \n\n with <p> tags)
                        // More sophisticated Markdown parsing could be added if needed.
                        const formattedText = rawText.replace(/\n\n/g, '</p><p class="mt-2">').replace(/\n/g, '<br>');
                        aiResponseArea.innerHTML = `<div class="prose prose-sm sm:prose dark:prose-invert max-w-none text-light-text dark:text-dark-text"><p>${formattedText}</p></div>`;
                    } else {
                        console.error('Unexpected API response structure:', result);
                        aiResponseArea.innerHTML = `<p class="text-red-500 dark:text-red-400">Sorry, I couldn't get a response. The AI might be busy or there was an unexpected issue. Please try again later.</p>`;
                    }
                } catch (error) {
                    console.error('Error calling Gemini API:', error);
                    aiResponseArea.innerHTML = `<p class="text-red-500 dark:text-red-400">Sorry, an error occurred while trying to reach the AI assistant: ${error.message}. Please try again later.</p>`;
                } finally {
                    // Reset loading state
                    aiButtonText.textContent = 'Ask Question';
                    aiLoadingSpinner.classList.add('hidden');
                    submitAiQuestionButton.disabled = false;
                }
            });
        }

    </script>
